	\draw [<->] 
	      (0, -1.5) 
	   -- (0,  1.5) node [pos=0.975, right] {$f(t)$};

    \draw [domain=-2:2, samples=1000]
          plot(\x, {sin(2 * pi * \x r)});

    \draw [domain=-2:2, samples=1000, red]
          plot(\x, {0.5 * sin(4 * pi * \x r)});

\end{tikzpicture}
